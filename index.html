	if(e&1) res=res*x%mod;
	return res;
}

EEE Dates or No Dates?

解法

离散化+差分。

首先将 Li/Ri/li/riL_i/R_i/l_i/r_iLi​/Ri​/li​/ri​ 合并后进行离散化处理。（注意，一定要合并！）

接着开两个差分数组 c1 和 c2，一个存 Li/RiL_i/R_iLi​/Ri​，一个存 li/ril_i/r_ili​/ri​。

Li/RiL_i/R_iLi​/Ri​ 直接对应存：++c1[L[i]],--c1[R[i]]；

li/ril_i/r_ili​/ri​ 则比较复杂：

  • forall i in [0,m): ++c2[r[i]],--c2[l[i+1]]；
  • ++c2[r[m]]。

如此，然后遍历整个数组，计算即可。

上代码：（std 总用时约 350ms350ms350ms — 202020 test cases）

#include <cstdio>
#include <cctype> // isdigit
#include <algorithm>
using namespace std;

int n, m, len, res = 0;
int L[100005], R[100005];
int l[100005], r[100005];
int allnum[400020];
int c1[400020], c2[400020], sum1, sum2, tmc;

inline int read() {
	int res=0; // no negative values
	char c=getchar();
	while(!isdigit(c)) c=getchar();
	while(isdigit(c)) res=(res<<3)+(res<<1)+(c^48),c=getchar();
	return res;
}

char __buf[5000000];

signed main() {
	setvbuf(stdout, __buf, _IOFBF, sizeof __buf);

	n=read();
	for(int i = 1; i <= n; ++i) {
		L[i]=read(), R[i]=read();
		allnum[i*2-1] = L[i];
		allnum[i*2] = R[i];
	}
	m=read();
	for(int i = 1; i <= m; ++i) {
		l[i]=read(), r[i]=read();
		allnum[n*2+i*2-1] = l[i];
		allnum[n*2+i*2] = r[i];
	}
	
	sort(allnum+1, allnum+n*2+m*2+1);
	len = unique(allnum+1, allnum+n*2+m*2+1)-allnum-1;

	for(int i = 1; i <= n; ++i) {
		L[i] = lower_bound(allnum+1, allnum+len+1, L[i])-allnum;
		R[i] = lower_bound(allnum+1, allnum+len+1, R[i])-allnum;
		++c1[L[i]];
		--c1[R[i]];
	}

	++c2[0];
	for(int i = 1; i <= m; ++i) {
		l[i] = lower_bound(allnum+1, allnum+len+1, l[i])-allnum;
		r[i] = lower_bound(allnum+1, allnum+len+1, r[i])-allnum;
		--c2[l[i]];
		++c2[r[i]];
	}
	--c2[len];

	sum1=sum2=tmc=0;
	for(int i = 0; i <= len; ++i) {
		sum1+=c1[i];
		sum2+=c2[i];
		if(sum1 && sum2) {
			tmc += (allnum[i+1]-allnum[i]);
		} else {
			res = max(res, tmc);
			tmc = 0;
		}
	}
	printf("%d\n", res);
	return 0;
}

FFF Date Number

解法

Trie 半模板题。（或者：01Trie，zxx 说这不是 Trie？）

Trie 是一棵字典树，在这棵树上可以方便地判断前缀和。

对于本题，我们可以把整数看做是一个二进制串，把这个串像字符串一样插入 Trie。

这时，Trie 类似一个二叉树，一个节点只有两个子节点，通向它们的边分别是 0 和 1。

为了不重复地查询，我们可以在插入一个数前先查询目前这个数与其他数 ⊕(xor⁡)\oplus(\operatorname{xor})⊕(xor) 的最大值。

如何查询？

  • 首先，将数插入 Trie 时，我们规定：从高位到低位依次插入（即，二进制串高位在前）；
  • 然后，查询时，从最高位开始：
    • (1) 如果在这一位上，有与当前位不同的一条边（即此边上的 0/10/10/1 值 ⊕\oplus⊕ 当前位后为 111），选择这条边，跳至 (3)；否则继续至 (2)。
    • (2) 选择与当前位相同的边，然后跳至 (3)。
    • (3) 进入下一位的查询，跳至 (1)。若已是最后一位，则退出，返回一路上所有边连起来后的值。

这样，一边读入，答案就出来了。时间复杂度 O(nlog⁡232)=O(32n)\mathcal{O}(n\log2^{32})=\mathcal{O}(32n)O(nlog232)=O(32n)。

上代码：（std 总用时约 1100ms1100ms1100ms — 202020 test cases）

#include <cstdio>
#include <cctype>
#include <cstring>
#include <utility>
#include <tuple>
using namespace std;

using uint = unsigned;
#define int uint

int n, q, a[100005];
int tot, trie[100005<<5][2];

void trie_init() {
	memset(trie, 0, sizeof trie);
	tot=1;
}
void trie_ins(int p) {
	int rt = 1;
	for(int i=(1<<31); i>0; i>>=1) {
		int q = (p&i)?1:0;
		if(trie[rt][q]==0) trie[rt][q]=++tot;
		rt = trie[rt][q];
	}
}
pair<int,int> trie_qry(int x) {
	int rt=1, ans1=0, ans2=0;
	for(int i=(1<<31); i>0; i>>=1) {
		int q = (x&i)?1:0;
		if(trie[rt][q^1]) {
			ans1 = ans1<<1|(q^1);
			ans2 = ans2<<1|1;
			rt = trie[rt][q^1];
		} else {
			ans1 = ans1<<1|q;
			ans2 = ans2<<1;
			rt = trie[rt][q];
		}
	}
	return {ans1,ans2};
}

char __buf[5000000];

signed main() {
	setvbuf(stdout, __buf, _IOFBF, sizeof __buf);
	scanf("%u", &n);
	trie_init();
	for(int i = 0; i < n; ++i) {
		scanf("%u", a+i);
		trie_ins(a[i]);
	}
	scanf("%u", &q);
	while(q--) {
		int x;
		scanf("%u", &x);
		int a, b;
		tie(a,b) = trie_qry(x);
		printf("%u %u\n", a, b);
	}
	return 0;
}

GGG Index Equation

题意解释

这题大家读题应该很困难吧。。。为了安慰大家脆弱的心灵，这里给出题意解释。

对于任意两个整数 xxx、yyy（x>yx>yx>y），一定能够找出一个实数 ddd，使得 (x)10=(y)d(x)_{10}=(y)_{d}(x)10​=(y)d​（即 101010 进制下的 xxx 等于 ddd 进制下的 yyy）。给出 xxx、yyy，找出这个实数 ddd，保留 444 位小数。

注意，上面明确说明 ddd 是实数！也就是说，ddd 不一定是整数。也就是说，本题不能进行枚举。

不枚举怎么办？

解法

题目中说明 xxx、yyy 不超过 444 位。让我们推一推式子：

设 x=a1a2a3a4‾x=\overline{a_1a_2a_3a_4}x=a1​a2​a3​a4​​，y=b1b2b3b4‾y=\overline{b_1b_2b_3b_4}y=b1​b2​b3​b4​​。

那么根据进制的定义，有 103a1+102a2+10a3+a4=d3b1+d2b2+db3+b410^3a_1+10^2a_2+10a_3+a_4=d^3b_1+d^2b_2+db_3+b_4103a1​+102a2​+10a3​+a4​=d3b1​+d2b2​+db3​+b4​。

整理一下，得 b1d3+b2d2+b3d=x−b4b_1d^3+b_2d^2+b_3d=x-b_4b1​d3+b2​d2+b3​d=x−b4​。

那么问题就成了求这个关于 ddd 的方程的一个正实数解。（进制默认为正数）

对于任何一元 nnn 次方程，未知数在第一象限上都是单调的。

那么二分 ddd 就行了。注意精度。

时间复杂度：O(log⁡x)\mathcal{O}(\log x)O(logx)。

上代码：（std 总用时约 30ms30ms30ms — 202020 test cases）

#include <cstdio>
#include <cmath>
#include <algorithm>
#include <cctype>
using namespace std;

const double eps = 1e-6;

int x, y;
int yy[10], len;

void inityy() {
	yy[1] = y / 1000;
	yy[2] = y / 100 % 10;
	yy[3] = y / 10 % 10;
	yy[4] = y % 10;
}

double f(double x) {
	return 1.0*yy[1]*x*x*x + 1.0*yy[2]*x*x + 1.0*yy[3]*x + 1.0*yy[4];
}

inline int read() {
	int res=0;
	char c=getchar();
	while(!isdigit(c)) c=getchar();
	while(isdigit(c)) res=(res<<3)+(res<<1)+(c^48),c=getchar();
	return res;
}

signed main() {
	x=read(),y=read();
	if(y<10) printf("-1\n");
	else {
		inityy();
		double l = 0.0, r = max(x, y);
		while(fabs(l-r)>eps) {
			double mid = (l+r)/2.0;
			double p = f(mid);
			if(p-x>eps) r = mid;
			else l = mid;
		}
		printf("%.5f\n", l);
	}
	return 0;
}

HHH Coding for Codes

本题应有两种做法。由于时间原因，这里仅讲解一种做法。

std 做法

由题目我们可以发现，在本题中，xxx 越小，ggg 越大；xxx 越大，ggg 越小。因此，xxx 满足单调性。

那么，既然要求 xxx 的个数，就可以求出 xxx 的最小值 xmaxx_{max}xmax​ 和最大值 xminx_{min}xmin​，然后输出 xmax−xmin+1x_{max}-x_{min}+1xmax​−xmin​+1 即可。

至于 xminx_{min}xmin​ 和 xmaxx_{max}xmax​，既然 xxx 单调，就可以用二分解答。

时间复杂度 O(tlog⁡x)\mathcal{O}(t\log x)O(tlogx)。

上代码：（std 总用时约 40ms40ms40ms — 101010 test cases）

#include <cstdio>
#include <cctype>
using namespace std;

#define div2(n) ((n)>>1)

long long t;
long long a, b, c, g;

long long check(long long p) {
	for(long long i = 0; i < g; ++i) {
		if(p>c) return i;
		p=p*a+b;
	}
	if(p>c) return g;
	return p;
}

inline long long read() {
	long long res=0;
	char c=getchar();
	while(!isdigit(c)) c=getchar();
	while(isdigit(c)) res=(res<<3)+(res<<1)+(c^48),c=getchar();
	return res;
}

signed main() {
	t=read();
	while(t--) {
		a=read(),b=read(),c=read(),g=read();
		long long ansl, ansr;
		
		if(g==0) {
			printf("%llu\n", (unsigned long long)(-1)-(unsigned long long)c);
			continue;
		}

		// CHECK MINIMUM
		long long l=0, r=c, x;
		while(l<r) {
			x = div2(l+r);
			long long f = check(x);
			if(f<=g) r = x;
			else l = x+1;
		}
		ansl = r;
		if(check(ansl)!=g) {
			printf("0\n");
			continue;
		}

		// CHECK MAXIMUM
		l=0, r=c;
		while(l<r) {
			x = div2(l+r+1);
			long long f = check(x);
			if(f<g) r = x-1;
			else l = x;
		}
		ansr = l;
		if(check(ansr)!=g) {
			printf("0\n");
			continue;
		}

		printf("%lld\n", ansr-ansl+1);
	}
	return 0;
}

n = int(input())
res = input()
a = [int(n) for n in res.split()]
mul = 1
for i in range(n):
    mul = mul*a[i]
res = 0
for i in range(n):
    res += (mul//a[i])
print(res)
